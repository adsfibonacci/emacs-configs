\subsection{Python}
\label{sec:org8ee049b}
This activates the Emacs communication with Pyenv Virtualenv.
In the local directories, it is able to automatically detect the proper pyrightconfig.json file, and use the server installed in the corresponding environment.
\begin{minted}[breaklines=true,fontsize=\scriptsize,frame=single]{elisp}
(unless (package-installed-p 'pyenv-mode)
  (package-install 'pyenv-mode))
(require 'pyenv-mode)
(pyenv-mode 1)

(defun pyenv-version-path ()
"Return the path to the python executable in the current pyenv/virtualenv."
(interactive)
(let* ((v-name (if (and (boundp 'pyenv-mode-version) pyenv-mode-version)
                   pyenv-mode-version
                 (string-trim (shell-command-to-string "pyenv version-name"))))
       (path (condition-case nil
                 (let ((root (string-trim 
                              (shell-command-to-string 
                               (concat "pyenv prefix " v-name)))))
                   (concat root "/bin/python"))
               (error "python3"))))
  (when (called-interactively-p 'interactive)
    (message "Resolved Python: %s" path))
  path))
\end{minted}

\begin{minted}[breaklines=true,fontsize=\scriptsize,frame=single]{elisp}
(defun eglot-python-contact (_)
  "Find the pyright-langserver. 
If remote, return the remote-local path (without TRAMP prefix).
If local, return the absolute path."
  (let* ((python-path (pyenv-version-path))
       (bin-dir (file-name-directory python-path))
       (pyright-bin (expand-file-name "pyright-langserver" bin-dir)))
  ;; We use file-local-name to strip the "/ssh:user@host:" prefix 
  ;; if it exists, because TRAMP process-start needs the remote-side path.
  (list (file-local-name pyright-bin) "--stdio")))  

(with-eval-after-load 'eglot
  (setq-default eglot-server-programs
                (assq-delete-all 'python-mode eglot-server-programs))
  (add-to-list 'eglot-server-programs `(python-mode . ,#'eglot-python-contact)))

(add-hook 'python-mode-hook #'eglot-ensure)
\end{minted}
